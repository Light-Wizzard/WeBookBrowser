100void XbelTreeReader::readXBEL()
101{

103
104 while (myXml.readNextStartElement())

106 if (myXml.name() == QLatin1String("folder")) { readFolder(0); }
107 else if (myXml.name() == QLatin1String("bookmark")) { readBookmark(0); }

178
179 while (myXml.readNextStartElement())

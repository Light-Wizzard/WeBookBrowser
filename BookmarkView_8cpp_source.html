29void BookmarkView::onCustomContextMenuRequested(const QPoint &pos)
30{
31 const QTreeWidgetItem *item = ui->treeWidget->itemAt(pos);

34 QMenu contextMenu;
35 QAction *copyAction = contextMenu.addAction(QObject::tr("Copy Link to Clipboard"));
36 QAction *openAction = contextMenu.addAction(QObject::tr("Open"));
37 QAction *addAction = contextMenu.addAction(QObject::tr("Add"));

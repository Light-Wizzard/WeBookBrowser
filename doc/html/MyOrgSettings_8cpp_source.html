625QString MyOrgSettings::getFileInfo(MyOrgSettings::MyFileinfo thisInfo, const QString &thisFileFolder)
626{
627 setMessage("get File Info", QString("getFileInfo(%1)").arg(thisFileFolder), Debug);
628 if (thisFileFolder.isEmpty()) { return ""; }
629 //
630 QFileInfo theFileInfo(thisFileFolder);
631 //
632 switch (thisInfo)
633 {
634 case AbsoluteFilePath:
635 // The absolute path name consists of the full path and the file name.
636 return theFileInfo.absoluteFilePath();
637 case AbsolutePath:
638 // Returns a file's path absolute path. This doesn't include the file name..
639 return theFileInfo.absolutePath();
640 case BaseName:
641 // Returns the base name of the file without the path.
642 return theFileInfo.baseName();
643 case BirthTime:
644 // Returns the date and time when the file was created / born.
645 return theFileInfo.birthTime().toString();
646 case CanonicalPath:
647 // Returns the file's path canonical path (excluding the file name), i.e. an absolute path without symbolic links or redundant "." or ".." elements.
648 return theFileInfo.canonicalPath();
649 case CanonicalFilePath:
650 // Returns the canonical path including the file name, i.e. an absolute path without symbolic links or redundant "." or ".." elements.
651 return theFileInfo.canonicalFilePath();
652 case CompleteBaseName:
653 // Returns the complete base name of the file without the path.
654 // "/tmp/archive.tar.gz" == "archive.tar"
655 return theFileInfo.completeBaseName();
656 case CompleteSuffix:
657 // Returns the complete suffix (extension) of the file.
658 // "/tmp/archive.tar.gz" == "tar.gz"
659 return theFileInfo.completeSuffix();
660 case Directory:
661 // Returns the path of the object's parent directory as a QDir object.
662 return theFileInfo.dir().currentPath();
663 case FileName:
664 // Returns the name of the file, excluding the path. getFileInfo(FileName, "")
665 return theFileInfo.fileName();
666 case FilePath:
667 // Returns the file name, including the path (which may be absolute or relative).
668 return theFileInfo.filePath();
669 case IsWritable:
670 // Returns true if the user can write to the file; otherwise returns false.
671 if (theFileInfo.isWritable()) return "true"; else return "false";
672 case FileSize:
673 return QString::number(theFileInfo.size());
674 case IsFile:
675 if (isFileExists(thisFileFolder)) return "true"; else return "false";
676 case IsFolder:
677 if (isPathExists(thisFileFolder)) return "true"; else return "false";
678 case IsSymLink:
679 if (theFileInfo.isSymLink()) return "true"; else return "false";
680 case SymLinkTarget:
681 if (theFileInfo.isSymLink()) return theFileInfo.symLinkTarget(); else return "";
682 }
683 return "";
684}

138void DownloadManagerWidget::updateWidget(int thisWidget)
139{
140 qreal totalBytes = myDownload.at(thisWidget)->totalBytes();
141 qreal receivedBytes = myDownload.at(thisWidget)->receivedBytes();
142 qreal bytesPerSecond = receivedBytes / myTimeAdded.at(thisWidget).elapsed() * 1000;
143
144 auto state = myDownload.at(thisWidget)->state();
145 QProgressBar *theProgressBar = qobject_cast<QProgressBar*>(myTableWidgetDownloads->cellWidget(thisWidget, CELL_PROGRESS));
146 QLabel *theTimerLable = qobject_cast<QLabel*>(myTableWidgetDownloads->cellWidget(thisWidget, CELL_TIMER));
147 theTimerLable->setText(QDateTime::fromMSecsSinceEpoch(myTimeAdded.at(thisWidget).elapsed()).toUTC().toString("hh:mm:ss"));
148 // State
149 switch (state)
150 {
151 case QWebEngineDownloadRequest::DownloadRequested:
152 Q_UNREACHABLE();
153 break;
154 case QWebEngineDownloadRequest::DownloadInProgress:
155 if (totalBytes >= 0)
156 {
157 theProgressBar->setValue(qRound(100 * receivedBytes / totalBytes));
158 theProgressBar->setDisabled(false);
159 theProgressBar->setFormat(tr("%p% - %1 of %2 downloaded - %3/s").arg(withUnit(receivedBytes), withUnit(totalBytes), withUnit(bytesPerSecond)));
160 }
161 else
162 {
163 theProgressBar->setValue(0);
164 theProgressBar->setDisabled(false);
165 theProgressBar->setFormat(tr("unknown size - %1 downloaded - %2/s").arg(withUnit(receivedBytes), withUnit(bytesPerSecond)));
166 }
167 break;
168 case QWebEngineDownloadRequest::DownloadCompleted:
169 theProgressBar->setValue(100);
170 theProgressBar->setDisabled(true);
171 theProgressBar->setFormat(tr("completed - %1 downloaded - %2/s").arg(withUnit(receivedBytes), withUnit(bytesPerSecond)));
172 break;
173 case QWebEngineDownloadRequest::DownloadCancelled:
174 theProgressBar->setValue(0);
175 theProgressBar->setDisabled(true);
176 theProgressBar->setFormat(tr("cancelled - %1 downloaded - %2/s").arg(withUnit(receivedBytes), withUnit(bytesPerSecond)));
177 break;
178 case QWebEngineDownloadRequest::DownloadInterrupted:
179 theProgressBar->setValue(0);
180 theProgressBar->setDisabled(true);
181 theProgressBar->setFormat(tr("interrupted: %1").arg(myDownload.at(thisWidget)->interruptReasonString()));
182 break;
183 }
184 QPushButton *theCancelButton = qobject_cast<QPushButton*>(myTableWidgetDownloads->cellWidget(thisWidget, CELL_CANCEL));
185 // FIXME remove on second click
186 if (state == QWebEngineDownloadRequest::DownloadInProgress)
187 {
188 static QIcon cancelIcon(QStringLiteral(":process-stop.png"));
189 theCancelButton->setIcon(cancelIcon);
190 theCancelButton->setToolTip(tr("Stop downloading"));
191 }
192 else
193 {
194 static QIcon removeIcon(QStringLiteral(":edit-clear.png"));
195 theCancelButton->setIcon(removeIcon);
196 theCancelButton->setToolTip(tr("Remove from list"));
197 }
198}

336inline QString DownloadManagerWidget::withUnit(qreal bytes)
337{
338 if (bytes < (1 << 10)) { return tr("%L1 B").arg(bytes); }
339 else if (bytes < (1 << 20)) { return tr("%L1 KiB").arg(bytes / (1 << 10), 0, 'f', 2); }
340 else if (bytes < (1 << 30)) { return tr("%L1 MiB").arg(bytes / (1 << 20), 0, 'f', 2); }
341 else { return tr("%L1 GiB").arg(bytes / (1 << 30), 0, 'f', 2); }
342}
